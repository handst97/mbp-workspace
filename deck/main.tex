\begin{frame}{Learning Objectives}
  By the end of this lecture, you will be able to:

\section{What is Data Science?}
% ============================================================================

\begin{frame}{The Data Science Revolution}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Data science} is an interdisciplinary field that uses:
      \begin{itemize}
        \item Scientific methods
        \item Algorithms and systems
        \item Domain expertise

      Data science bridges the gap between raw data and
      actionable business decisions.

\section{The Data Science Workflow}
% ============================================================================

\begin{frame}{The Data Science Pipeline}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\textwidth]{fig_pipeline.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Key observations:}

      \vspace{0.5em}
      \begin{itemize}
        \item \alert{Up to 40\% on cleaning}---real data is messy
        \item Modeling is only 15\% of the work
        \item Reporting often undervalued
      \end{itemize}

      \vspace{0.5em}
      \begin{exampleblock}{Best Practice}
        Invest time in understanding your data before modeling.
        ``Garbage in, garbage out.''
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{From Description to Prescription}

\begin{frame}{Descriptive and Diagnostic Analytics}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Descriptive Analytics}\\
      \textit{What happened?}
      \begin{itemize}
        \item Dashboards and reports
        \item Historical summaries
        \item KPI tracking
      \end{itemize}

      \vspace{0.3em}
      \textit{Example: Monthly sales report showing revenue by region}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Diagnostic Analytics}\\
      \textit{Why did it happen?}
      \begin{itemize}
        \item Root cause analysis
        \item Correlation studies
        \item Drill-down analysis
      \end{itemize}

      \vspace{0.3em}
      \textit{Example: Why did sales drop 15\% in Q3?}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Predictive and Prescriptive Analytics}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Predictive Analytics}\\
      \textit{What will happen?}
      \begin{itemize}
        \item Forecasting models
        \item Risk assessment
        \item Trend analysis
      \end{itemize}

      \vspace{0.3em}
      \textit{Example: Customer churn prediction model}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Prescriptive Analytics}\\
      \textit{What should we do?}
      \begin{itemize}
        \item Optimization algorithms
        \item Decision support systems
        \item Automated recommendations
      \end{itemize}

      \vspace{0.3em}
      \textit{Example: Dynamic pricing engine}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Example: Predicting Student Performance}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\textwidth]{fig_sample_analysis.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Research question:}\\
      Does study time predict exam performance?

      \vspace{0.5em}
      \textbf{Method:}
      \begin{itemize}
        \item Simple linear regression
        \item 80 student observations
        \item Weekly hours vs.\ exam score
      \end{itemize}

      \vspace{0.3em}
      \textbf{Finding:}\\
      Strong positive correlation ($R^2 \approx 0.6$); each additional hour
      $\approx$ 5 points.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The Bias-Variance Tradeoff}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\textwidth]{fig_bias_variance.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Bias} (underfitting):
      \begin{itemize}
        \item Model too simple
        \item Misses important patterns
      \end{itemize}

      \vspace{0.3em}
      \textbf{Variance} (overfitting):
      \begin{itemize}
        \item Model too complex
        \item Fits noise in training data
      \end{itemize}

      \vspace{0.3em}
      \begin{alertblock}{Goal}
        Find the \textbf{optimal complexity} that minimizes total error.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Summary}
% ============================================================================

\begin{frame}{Key Takeaways}
  \begin{enumerate}
    \item Data science combines statistics, computing, and domain expertise
    \item The workflow: collect $\to$ clean $\to$ analyze $\to$
          model $\to$ report
    \item Analytics maturity: descriptive $\to$ diagnostic $\to$
          predictive $\to$ prescriptive
    \item Machine learning: supervised, unsupervised, reinforcement
    \item Balance bias and variance for optimal models
  \end{enumerate}
